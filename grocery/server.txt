import socket
s=socket.socket()
host=socket.gethostname()
port=6666
s.bind((host,port))
s.listen(11)
print("waiting for response")
cl,addr=s.accept()
while True:
    print("connected with client",addr)
    command=input("enter command:")
    cl.send(bytes(command,'utf-8'))
    k=cl.recv(1024).decode()
    print('recieved from client: '+k)
